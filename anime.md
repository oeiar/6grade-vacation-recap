## Scripts for ani

### min:sec adder

```js
function a(x) {
  let xs = x
    .split("\n")
    .map((i) => i.replace("-", "").replaceAll(" ", "").split(":"));
  let sec = 0;
  let min = 0;
  xs.forEach((i) => (sec += Number(i[1])));
  xs.forEach((i) => (min += Number(i[0])));
  min += Math.floor(sec / 60);
  sec %= 60;
  return [min, sec];
}
```

function b(x) {
